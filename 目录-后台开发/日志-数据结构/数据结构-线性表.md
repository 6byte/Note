### ArrayList简单实现

```java

public class Main {
	private int[] data;
	private int size;

//	有参构造函数
	public Main(int capacity) {
//		初始化数组容量		capacity
		data = new int[capacity];
//		相当于下标
		size = 0;
	}

//	无参构造函数
	public Main() {
		this(10);
	}

//	用户查看数组元素个数
	public int getSize() {
		return data.length;
	}

//	用户查看数组容量
	public int getCapacity() {
		return size;
	}

//	查看数组是否为空
	public boolean isEmpty() {
		return size == 0;
	}

//	向数组添加元素
	public void addElement(int e) {
		if (size < this.getSize()) {
			this.data[size++] = e;
		}

	}

	// 将元素插入指定位置
	public void insertElement(int index, int e) {
		if(index < 0 || index > this.getSize()) {
			throw new IllegalArgumentException("Add failured , you put into a IllegalArgument");
		}
		else {
            //data[i+1] = data[i]	将index之前的元素往后移一位
            //i < index;			在index位置停下，让出一个空位，留给e元素		
			for (int i = size-1; i < index; i++) {
				data[i+1] = data[i];
			}
			data[index] = e;
			size++;
		}
	}

	public static void main(String[] args) {

	}

}

```
